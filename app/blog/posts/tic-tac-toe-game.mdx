Design a Tic-Tac-Toe game for two players that allows them to play on a 3×3 grid, 
take alternate turns, validate moves, and determine the winner or a draw based on the game state.

![tic-tac-toe](/tic-tac-toe/tic-tac-toe.png)

---

### Functional Requirements

1. Assign each player a unique symbol (e.g., X and O) at the start of the game.

2. Enforce alternating turns between players and always indicate the current player's turn.

3. Allow players to make a move by selecting a cell (row, column), 
ensuring the cell is not already occupied, and update the board accordingly.

4. Allow players to undo the last move, restoring the previous board state and 
switching the turn back to the respective player.

5. Check after every move whether the current player has won by completing a row, column, or diagonal.

6. Detect a draw when all cells are filled and no player has won.

7. Keep track of the score for each player.

### Primary Use Cases

The primary use cases involved in the Tic-Tac-Toe game include players making moves on the board and 
optionally undoing previous moves, with the system responsible for maintaining board state, 
enforcing turn order, and determining game outcomes such as win or draw.

#### Use Case 1: Move a New Move

1. A player makes a move by selecting a cell using a `[row, column]` coordinate 
on the board.

2. The system validates that the selected cell is within bounds and not already occupied. 
If valid, it updates the cell with the player's symbol.

3. Next, the system evaluates whether the move results in a win or draw. If the game is still 
in progress, the turn is switched to the next player.

4. If a player wins, the system updates the player's score to reflect the victory.

#### Use Case 2: Undo a Previous Move

1. A player requests to undo the last move.

2. The system retrieves the most recent move from the move history and reverts the board state 
by clearing that move.

3. Next, the system switches the turn back to the player who made the undone move and reverts 
the game state from WIN / DRAW to IN-PROGRESS if needed.

```java
public class Game {
    Board board;
    Player[] players;
    int currentPlayerIndex;
    GameStatus status;
    Stack<Move> history;

    // For O(1) winner check
    int[] rows;
    int[] cols;
    int diagonal;
    int antiDiagonal;

    Game(int size, Player p1, Player p2) {
        this.board = new Board(size);
        this.players = new Player[]{p1, p2};
        this.currentPlayerIndex = 0;
        this.status = GameStatus.IN_PROGRESS;
        this.history = new Stack<>();

        this.rows = new int[size];
        this.cols = new int[size];
        this.diagonal = 0;
        this.antiDiagonal = 0;
    }

    public void playMove(int row, int col) {

        Player player = players[currentPlayerIndex];

        if (!board.isValidMove(row, col)) {
            System.out.println("Invalid move!");
            return;
        }

        board.makeMove(row, col, player.symbol);
        history.push(new Move(row, col, player));

        updateCounts(row, col, player.symbol);

        if (checkWinner(row, col)) {
            status = GameStatus.WIN;
            board.printBoard();
            System.out.println(player.name + " wins!");
            return;
        }

        if (board.isFull()) {
            status = GameStatus.DRAW;
            board.printBoard();
            System.out.println("Game is a draw!");
            return;
        }

        switchTurn();
        board.printBoard();
    }

    public void undo() {
        if (history.isEmpty()) {
            System.out.println("No moves to undo!");
            return;
        }

        Move lastMove = history.pop();
        board.undoMove(lastMove.row, lastMove.col);

        revertCounts(lastMove.row, lastMove.col, lastMove.player.symbol);

        currentPlayerIndex = (currentPlayerIndex + 1) % 2;
        status = GameStatus.IN_PROGRESS;

        board.printBoard();
        System.out.println("Undo successful!");
    }

    private void switchTurn() {
        currentPlayerIndex = (currentPlayerIndex + 1) % 2;
    }

    private void updateCounts(int row, int col, Symbol symbol) {
        int val = (symbol == Symbol.X) ? 1 : -1;

        rows[row] += val;
        cols[col] += val;

        if (row == col) diagonal += val;
        if (row + col == board.size - 1) antiDiagonal += val;
    }

    private void revertCounts(int row, int col, Symbol symbol) {
        int val = (symbol == Symbol.X) ? 1 : -1;

        rows[row] -= val;
        cols[col] -= val;

        if (row == col) diagonal -= val;
        if (row + col == board.size - 1) antiDiagonal -= val;
    }

    private boolean checkWinner(int row, int col) {
        int size = board.size;

        return Math.abs(rows[row]) == size ||
               Math.abs(cols[col]) == size ||
               Math.abs(diagonal) == size ||
               Math.abs(antiDiagonal) == size;
    }
}
```

```java
// ENUMS
enum Symbol {
    X, O, EMPTY
}

enum GameStatus {
    IN_PROGRESS, WIN, DRAW
}

// CELL
class Cell {
    int row;
    int col;
    Symbol symbol;

    Cell(int row, int col) {
        this.row = row;
        this.col = col;
        this.symbol = Symbol.EMPTY;
    }
}

// MOVE
class Move {
    int row;
    int col;
    Player player;

    Move(int row, int col, Player player) {
        this.row = row;
        this.col = col;
        this.player = player;
    }
}

// PLAYER
class Player {
    String name;
    Symbol symbol;

    Player(String name, Symbol symbol) {
        this.name = name;
        this.symbol = symbol;
    }
}

// BOARD
class Board {
    int size;
    Cell[][] grid;
    int movesCount;

    Board(int size) {
        this.size = size;
        this.grid = new Cell[size][size];
        this.movesCount = 0;

        for (int i = 0; i < size; i++) {
            for (int j = 0; j < size; j++) {
                grid[i][j] = new Cell(i, j);
            }
        }
    }

    boolean isValidMove(int row, int col) {
        return row >= 0 && row < size &&
               col >= 0 && col < size &&
               grid[row][col].symbol == Symbol.EMPTY;
    }

    void makeMove(int row, int col, Symbol symbol) {
        grid[row][col].symbol = symbol;
        movesCount++;
    }

    void undoMove(int row, int col) {
        grid[row][col].symbol = Symbol.EMPTY;
        movesCount--;
    }

    boolean isFull() {
        return movesCount == size * size;
    }

    void printBoard() {
        for (int i = 0; i < size; i++) {
            for (int j = 0; j < size; j++) {
                System.out.print(grid[i][j].symbol + " ");
            }
            System.out.println();
        }
        System.out.println();
    }
}

public class TicTacToe {
    public static void main(String[] args) {
        Player p1 = new Player("Player 1", Symbol.X);
        Player p2 = new Player("Player 2", Symbol.O);

        Game game = new Game(3, p1, p2);

        game.playMove(0, 0);
        game.playMove(0, 1);
        game.playMove(1, 1);
        game.playMove(0, 2);
        game.playMove(2, 2); // win

        game.undo(); // undo win
        game.playMove(2, 0);
    }
}
```
